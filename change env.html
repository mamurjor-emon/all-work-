/** * ENV Data Set * @param array $data * @return \Illuminate\Http\Response */ protected function changeEnvData(array $data) { if (count($data) > 0) { $env = file_get_contents(base_path() . '/.env'); $env = preg_split('/\s+/', $env); foreach ($data as $key => $value) { foreach ($env as $env_key => $env_value) { $entry = explode("=", $env_value, 2); if ($entry[0] == $key) { $env[$env_key] = $key . "=" . $value; } else { $env[$env_key] = $env_value; } } } $env = implode("\n", $env); file_put_contents(base_path() . '/.env', $env); return true; } else { return false; } } // env set value $this->changeEnvData([ 'MAIL_MAILER' => $request->mail_mailer, 'MAIL_HOST' => $request->mail_host, 'MAIL_PORT' => $request->mail_port, 'MAIL_USERNAME' => $request->mail_username, 'MAIL_PASSWORD' => $request->mail_password, 'MAIL_ENCRYPTION' => $request->mail_encryption, 'MAIL_FROM_ADDRESS' => $request->mail_from_address ]);
