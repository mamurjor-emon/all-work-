public function index(Request $request) { if($request->all() == []){ $salesProducts = Sale::latest('id')->paginate(25); }else{ if ($request->start_date) { $today_date = $request->start_date; } else { $today_date = now()->format('Y-m-d'); } if ($request->end_date) { $last_date = $request->end_date; } else { $last_date = now()->format('Y-m-d'); } $salesProducts = Sale::whereDate('created_at', '>=', $today_date)->whereDate('created_at', '<=', $last_date)->paginate(25); } }
